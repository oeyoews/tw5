\define startup_actions()
<$importvariables filter="TiddlyTools/Time/Alarms"> <<alarms_startup>> </$importvariables>
\end

\define ticker_actions()
<$action-setfield $tiddler="$:/temp/time/ticker" text=<<now [UTC]YYYY0MM0DD0hh0mm0ss0XXX>> />
<$importvariables filter="TiddlyTools/Time/Alarms">    <<alarms_tick>>    </$importvariables>
<$importvariables filter="TiddlyTools/Time/AutoSaver"> <<autosaver_tick>> </$importvariables>
<$importvariables filter="TiddlyTools/Time/CountDown"> <<countdown_tick>> </$importvariables>
\end

<$action-timeout    delay="1000" actions=<<startup_actions>>/>
<$action-timeout interval="1000" actions=<<ticker_actions>> />
